190 function line(ctx, x1, y1, x2, y2) {
191  ctx.beginPath();
192  ctx.moveTo(x1, y1);
193  ctx.lineTo(x2, y2);
194  ctx.stroke();
195 }
196 )");
197  }
198  virtual void draw(std::ostream& os) const override {
199  if(points.size() == 2) {
200  os << "line(ctx, " << static_cast<int>(points[0].x) << ", " << static_cast<int>(points[0].y)
201  << ", " << static_cast<int>(points[1].x) << ", " << static_cast<int>(points[1].y) << ");\n";
202  }
203  else {
204  os << "ctx.beginPath();\n";
205  os << "ctx.moveTo(" << static_cast<int>(points[0].x) << ", " << static_cast<int>(points[0].y) << ");\n";
206  for(size_t p_i = 1; p_i < points.size(); ++p_i) {
207  os << "ctx.lineTo(" << static_cast<int>(points[p_i].x) << ", " << static_cast<int>(points[p_i].y) << ");\n";
208  }
209  if(close_path)
210  os << "ctx.closePath();\n";
211  os << (fill ? "ctx.fill();\n" : "ctx.stroke();\n");

514  os << "<script>\n";

517  os << "const num_frames = " << get_num_frames() << ";\n";

519  os << "var wait_counter = 0;\n";
520  os << "var num_wait_frames = " << wait_frames << ";\n";
521  os << "var no_clear = " << (no_clear ? "true" : "false") << ";\n";
522

524 
525  write_back_and_foreground(os);
526  write_frames(os);
527 
528  os << R"(
529 window.onload = function() {
530  ctx = canvas.getContext('2d'),
531  (function drawFrame () {
532  if(frame_counter == 0 || !no_clear)
533  ctx.clearRect(0, 0, canvas.width, canvas.height);
534  draw_bkgnd(ctx);
535  (frames[frame_counter])(ctx);
536  draw_frgnd(ctx);
537  wait_counter = (num_wait_frames == 0) ? 0 : ((wait_counter + 1) % num_wait_frames);
538  if(wait_counter == 0)
539  frame_counter = (frame_counter + 1) % num_frames;
540  window.requestAnimationFrame(drawFrame, canvas);
541  }());
542 }
543 
544 //-->
545 </script>
546 <noscript>JavaScript is required to display this content.</noscript>
547 )";
